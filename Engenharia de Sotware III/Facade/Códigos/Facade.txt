// Subsistemas (Subsystems)

class CartaoCredito {
    public void processarPagamento(double valor) {
        System.out.println("Pagamento com cartão de crédito no valor de R$" + valor);
    }
}

class CartaoDebito {
    public void processarPagamento(double valor) {
        System.out.println("Pagamento com cartão de débito no valor de R$" + valor);
    }
}

class Dinheiro {
    public void processarPagamento(double valor) {
        System.out.println("Pagamento em dinheiro no valor de R$" + valor);
    }
}

// Facade

class PagamentoFacade {
    private CartaoCredito cartaoCredito;
    private CartaoDebito cartaoDebito;
    private Dinheiro dinheiro;

    public PagamentoFacade() {
        this.cartaoCredito = new CartaoCredito();
        this.cartaoDebito = new CartaoDebito();
        this.dinheiro = new Dinheiro();
    }

    public void processarPagamentoCartaoCredito(double valor) {
        cartaoCredito.processarPagamento(valor);
    }

    public void processarPagamentoCartaoDebito(double valor) {
        cartaoDebito.processarPagamento(valor);
    }

    public void processarPagamentoDinheiro(double valor) {
        dinheiro.processarPagamento(valor);
    }
}

// Cliente

public class Main {
    public static void main(String[] args) {
        PagamentoFacade pagamentoFacade = new PagamentoFacade();

        // Utilizando a Facade para processar pagamentos
        pagamentoFacade.processarPagamentoCartaoCredito(100.0);
        pagamentoFacade.processarPagamentoCartaoDebito(50.0);
        pagamentoFacade.processarPagamentoDinheiro(30.0);
    }
}
